import csv

file_name = 'students.csv'
header = ['name', 'age', 'grade']
data = [
    ['Alice', 20, 'A'],
    ['Bob', 21, 'B'],
    ['Charlie', 19, 'C']
]

with open(file_name, mode='w', newline='') as file:
    writer = csv.writer(file)
    writer.writerow(header)
    writer.writerows(data)

print(f"File '{file_name}' created successfully with 3 records.")

import csv

file_name = 'sales.csv'
total_sales = 0

try:
    with open(file_name, mode='r', newline='') as file:
        reader = csv.reader(file)
        next(reader, None) # Skip header
        for row in reader:
            if len(row) == 3:
                try:
                    price = float(row[1])
                    quantity = int(row[2])
                    total_sales += price * quantity
                except ValueError:
                    print(f"Skipping row due to invalid data: {row}")

    print(f"Total sales: ${total_sales:.2f}")
except FileNotFoundError:
    print(f"Error: The file '{file_name}' was not found.")

OUTPUT:
Total sales: $6450.00

import csv

file_name = 'people.csv'

try:
    with open(file_name, mode='r', newline='') as file:
        reader = csv.reader(file)
        header = next(reader, None)
        if header:
            print(header) # Print header
            for row in reader:
                if len(row) > 1:
                    try:
                        if int(row[1]) > 30:
                            print(row)
                    except ValueError:
                        continue # Skip rows with non-integer age

except FileNotFoundError:
    print(f"Error: The file '{file_name}' was not found.")

OUTPUT:
['name', 'age', 'city']
['Bob', '35', 'Los Angeles']
['Charlie', '40', 'Chicago']
['Eve', '31', 'Miami']

import csv

source_file = 'source.csv'
copy_file = 'copy.csv'

try:
    with open(source_file, mode='r', newline='') as src, \
         open(copy_file, mode='w', newline='') as dst:
        reader = csv.reader(src)
        writer = csv.writer(dst)
        for row in reader:
            writer.writerow(row)

    print(f"All contents copied from '{source_file}' to '{copy_file}'.")
except FileNotFoundError:
    print(f"Error: The source file '{source_file}' was not found.")

OUTPUT:
All contents copied from 'source.csv' to 'copy.csv'.
